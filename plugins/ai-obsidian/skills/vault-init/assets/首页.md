// OKLCH 色环等间距取色(L=0.58, C=0.14, H 均分)——最大化视觉区分
function oklchToHex(L, C, H){
  const a = C * Math.cos(H * Math.PI / 180), b = C * Math.sin(H * Math.PI / 180);
  const l_ = L + 0.3963377774*a + 0.2158037573*b;
  const m_ = L - 0.1055613458*a - 0.0638541728*b;
  const s_ = L - 0.0894841775*a - 1.2914855480*b;
  const lr = l_*l_*l_, mr = m_*m_*m_, sr = s_*s_*s_;
  const R = 4.0767416621*lr - 3.3077115913*mr + 0.2309699292*sr;
  const G = -1.2684380046*lr + 2.6097574011*mr - 0.3413193965*sr;
  const B = -0.0041960863*lr - 0.7034186147*mr + 1.7076147010*sr;
  const f = c => { c = Math.max(0, Math.min(1, c)); const s = c <= 0.0031308 ? 12.92*c : 1.055*Math.pow(c, 1/2.4)-0.055; return Math.round(s*255).toString(16).padStart(2,"0"); };
  return "#" + f(R) + f(G) + f(B);
}

function ago(dt){
  const d = now - dt.toMillis();
  if (d < DAY) return "今天";
  const n = Math.floor(d / DAY);
  if (n < 30) return n + "天前";
  const mo = Math.floor(n / 30);
  if (mo < 12) return mo + "个月前";
  return Math.floor(mo / 12) + "年前";
}
function ymd(t){ const d = new Date(t); return d.getFullYear() + "-" + (d.getMonth()+1) + "-" + d.getDate(); }
const perDay = {};
for (const p of arr){ const k = ymd(ms(p)); perDay[k] = (perDay[k] || 0) + 1; }
function countInRange(startDaysAgo, endDaysAgo){
  let c = 0;
  for (const p of arr){ const age = (now - ms(p)) / DAY; if (age >= endDaysAgo && age < startDaysAgo) c++; }
  return c;
}

function attachTip(el, text){
  el.setAttribute("aria-label", text);
  el.addEventListener("mouseenter", () => { tip.setText(text); tip.style.opacity = "1"; });
  el.addEventListener("mousemove", (e) => { tip.style.left = (e.clientX + 12) + "px"; tip.style.top = (e.clientY + 14) + "px"; });
  el.addEventListener("mouseleave", () => { tip.style.opacity = "0"; });
}

function box(parent, css){ const d = parent.createDiv(); if (css) d.style.cssText = css; return d; }

function miniBars(parent, values, opt){
  opt = opt || {};
  const h = opt.height || 34, color = opt.color || BRAND, labels = opt.labels || [];

  const row = box(parent, `display:flex;align-items:flex-end;gap:3px;height:${h}px;`);
  values.forEach((v, i) => {
    const pct = Math.round(v / max * 100);

    b.addClass("cx-hit");
    if (labels[i]) attachTip(b, labels[i] + " · " + v + " 次更新");
  });
  return row;
}
